USAGE EXAMPLE
================================================================================

from services.export_service import export_resume_to_pdf, export_resume_to_docx

resume = {
    "name": "Jane Smith",
    "email": "jane@example.com",
    "phone": "+1-555-0123",
    "skills": ["Python", "FastAPI", "AWS"],
    "experience": [...],
    "education": [...]
}

# Export to PDF
pdf_bytes = export_resume_to_pdf(resume)
with open("resume.pdf", "wb") as f:
    f.write(pdf_bytes)

# Export to DOCX
docx_bytes = export_resume_to_docx(resume)
with open("resume.docx", "wb") as f:
    f.write(docx_bytes)
